function hitSound(){ if(!audioCtx) return; let o=audioCtx.createOscillator(); let g=audioCtx.createGain(); o.type="triangle"; o.frequency.value=180; o.connect(g); g.connect(audioCtx.destination); g.gain.setValueAtTime(0.4,audioCtx.currentTime); g.gain.exponentialRampToValueAtTime(0.001,audioCtx.currentTime+0.25); o.start(); o.stop(audioCtx.currentTime+0.25); } function alarmSound(){ if(!audioCtx) return; let o=audioCtx.createOscillator(); let g=audioCtx.createGain(); o.type="square"; o.frequency.value=320; o.connect(g); g.connect(audioCtx.destination); g.gain.setValueAtTime(0.5,audioCtx.currentTime); g.gain.exponentialRampToValueAtTime(0.001,audioCtx.currentTime+0.6); o.start(); o.stop(audioCtx.currentTime+0.6); }
